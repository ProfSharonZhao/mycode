  <h1>Stress and Performance</h1>

  <div class="controls">

    <button onclick="adjustStress(1)">Increase Stress</button>

    <button onclick="adjustStress(-1)">Decrease Stress</button>

    <button onclick="resetValues()">Reset</button>

  </div>

  <div class="flowchart">

    <div class="box">

      <strong>Stress</strong>

      <canvas id="stressChart"></canvas>

    <div class="row">

      <div class="box">

        <strong>Motivation</strong>

        <canvas id="motivationChart"></canvas>

      <div class="box">

      </div>

    <div class="box">

      <strong>Performance</strong>

      <canvas id="performanceChart"></canvas>

    </div>

  <script>

    let stressLevel = 5;

    let motivationLevel = 7;

    let anxietyLevel = 3;

    let performanceLevel = 7;


    const stressCtx = document.getElementById('stressChart').getContext('2d');

    const motivationCtx = document.getElementById('motivationChart').getContext('2d');

    const anxietyCtx = document.getElementById('anxietyChart').getContext('2d');

    const performanceCtx = document.getElementById('performanceChart').getContext('2d');


    const stressChart = new Chart(stressCtx, {

      type: 'bar',

      data: {

        labels: ['Stress'],

        datasets: [{

          label: 'Level',

          data: [stressLevel],

          backgroundColor: 'orange'

        }]

      },

      options: { scales: { y: { min: 0, max: 10 } } }

    });


    const motivationChart = new Chart(motivationCtx, {

      type: 'bar',

      data: {

        labels: ['Motivation'],

        datasets: [{

          label: 'Level',

          data: [motivationLevel],

          backgroundColor: 'blue'

        }]

      },

      options: { scales: { y: { min: 0, max: 10 } } }

    });


    const anxietyChart = new Chart(anxietyCtx, {

      type: 'bar',

      data: {

        labels: ['Anxiety'],

        datasets: [{

          label: 'Level',

          data: [anxietyLevel],

          backgroundColor: 'purple'

        }]

      },

      options: { scales: { y: { min: 0, max: 10 } } }

    });


    const performanceChart = new Chart(performanceCtx, {

      type: 'bar',

      data: {

        labels: ['Performance'],

        datasets: [{

          label: 'Level',

          data: [performanceLevel],

          backgroundColor: 'green'

        }]

      },

      options: { scales: { y: { min: 0, max: 10 } } }

    });


    function adjustStress(delta) {

      stressLevel = Math.max(0, Math.min(10, stressLevel + delta));

      motivationLevel = -Math.pow(stressLevel - 4, 2) / 2.5 + 10; // peak at moderate stress

      anxietyLevel = Math.pow(stressLevel / 10, 2) * 10; // increases with stress

      performanceLevel = Math.max(0, Math.min(10, (motivationLevel - anxietyLevel / 2)));


      updateCharts();

    }


    function resetValues() {

      stressLevel = 5;

      motivationLevel = 7;

      anxietyLevel = 3;

      performanceLevel = 7;

      updateCharts();

    }


    function updateCharts() {

      stressChart.data.datasets[0].data[0] = stressLevel;

      motivationChart.data.datasets[0].data[0] = Math.round(motivationLevel * 10) / 10;

      anxietyChart.data.datasets[0].data[0] = Math.round(anxietyLevel * 10) / 10;

      performanceChart.data.datasets[0].data[0] = Math.round(performanceLevel * 10) / 10;


      stressChart.update();

      motivationChart.update();

      anxietyChart.update();

      performanceChart.update();

    }


    updateCharts();

  </script>
