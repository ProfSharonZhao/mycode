  <script>

    let stress = 1;

    let focus = 5;

    let anxiety = 1;

    let performanceLevel = 5;


    const charts = {};


    function createChart(ctxId, label, value, color) {

      const ctx = document.getElementById(ctxId);

      charts[ctxId] = new Chart(ctx, {

        type: 'bar',

        data: {

          labels: [label],

          datasets: [{

            label: label,

            data: [value],

            backgroundColor: color,

            borderColor: color.replace('0.6', '1'),

            borderWidth: 1

          }]

        },

        options: {

          animation: {

            duration: 800

          },

          responsive: true,

          plugins: {

            legend: { display: false },

            title: { display: false }

          },

          scales: {

            y: {

              beginAtZero: true,

              max: 10

            },

            x: {

              display: false

            }

          }

        }

      });

    }


    createChart('stressChart', 'Stress', stress, 'rgba(255, 99, 132, 0.6)');

    createChart('focusChart', 'Focus', focus, 'rgba(54, 162, 235, 0.6)');

    createChart('anxietyChart', 'Anxiety', anxiety, 'rgba(255, 206, 86, 0.6)');

    createChart('performanceChart', 'Performance', performanceLevel, 'rgba(75, 192, 192, 0.6)');


    function updateAllCharts() {

      charts['stressChart'].data.datasets[0].data = [stress];

      charts['stressChart'].update();


      charts['focusChart'].data.datasets[0].data = [focus];

      charts['focusChart'].update();


      charts['anxietyChart'].data.datasets[0].data = [anxiety];

      charts['anxietyChart'].update();


      charts['performanceChart'].data.datasets[0].data = [performanceLevel];

      charts['performanceChart'].update();

    }


    function increaseStress() {

      if (stress < 10) {

        stress++;

        if (stress <= 4) {

          focus = Math.min(focus + 1, 10);

          anxiety = Math.min(anxiety + 1, 10);

          performanceLevel = Math.min(performanceLevel + 1, 10);

        } else {

          focus = Math.max(focus - 1, 0);

          anxiety = Math.min(anxiety + 1, 10);

          performanceLevel = Math.max(performanceLevel - 1, 0);

        }

        updateAllCharts();

      }

    }


    function resetCharts() {

      stress = 1;

      focus = 5;

      anxiety = 1;

      performanceLevel = 5;

      updateAllCharts();

    }
